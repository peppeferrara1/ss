  1. Definisce un oggetto config che contiene le impostazioni per il grafico.

  4. Le opzioni del grafico sono definite nell'oggetto options.

Dove mettere questa parte di codice dipende da come si sta sviluppando l'applicazione web. typescript
import { Component, OnInit } from '@angular/core';
import { Chart } from 'chart.js';
import { DataService } from 'src/app/services/data.service';

@Component({
  selector: 'app-product-rreport',
  templateUrl: './product-rreport.component.html',
  styleUrls: ['./product-rreport.component.css']
})

  constructor(private dataService: DataService) { }

      this.chartData = data;
      this.createChart();
    });

      data: {

          {
            label: "Sales",

            backgroundColor: 'blue'
          },
          {
            label: "Profit",
            data: this.chartData.profitData,
            backgroundColor: 'limegreen'
          }
        ]
      },
      options: {
        aspectRatio: 2.5
      }
    });
  }
}
